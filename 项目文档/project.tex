\chapter{项目概述}

\section{项目背景}

本项目基于 ESP32-S3 微控制器，旨在开发一套具备 AI 联网对话能力和无线手柄控制功能的智能交互系统。ESP32-S3 以其强大的双核处理器、丰富的外设接口（I2S、SPI、I2C、UART 等）以及内置 Wi-Fi 和 BLE 功能，非常适合作为本项目的主控芯片。

\section{项目目标}

\begin{enumerate}[leftmargin=2em]
    \item \textbf{AI 联网对话}：通过硅麦（PDM 数字麦克风）采集语音，上传至云端进行语音识别（ASR），调用大语言模型（LLM）API 生成回复，再通过 TTS 语音合成后经扬声器播放，实现完整的语音对话闭环。
    \item \textbf{无线手柄}：基于 Wi-Fi（UDP/TCP）通信，实现外接手柄与 ESP32-S3 之间的无线数据传输，用于控制 PWM 显示器等外设。
    \item \textbf{屏幕显示}：在 OLED 或 LCD 屏幕上实时显示对话内容、手柄状态等信息。
\end{enumerate}

\section{阶段三：语音采集（硅麦驱动）}
\textbf{预估：2 天 \quad 优先级：$\star\star\star\star\star$}

这是当前任务的\textbf{核心难点}，需要重点关注。

\subsubsection*{子任务清单}

\begin{enumerate}[leftmargin=2em, label=\textbf{3.\arabic*}]
    \item \textbf{硬件连接}
    \begin{itemize}[leftmargin=3em]
        \item 确认硅麦引脚定义（DATA、CLK、L/R、VDD、GND）
        \item 连接至 ESP32-S3 的 I2S 接口
        \item \texttt{注意：ESP32-S3 的 I2S 引脚与经典 ESP32 有所不同，务必查阅引脚定义表}
    \end{itemize}
    
    \item \textbf{I2S 驱动配置}
    \begin{itemize}[leftmargin=3em]
        \item 使用 ESP-IDF 的 \texttt{I2S} 驱动库或 Arduino 的 \texttt{I2SSampler} 库
        \item 配置采样率（建议 16kHz）、位深（16bit）、通道数（单声道）
        \item 配置 PDM 模式（硅麦通常是 PDM 输出）
    \end{itemize}
    
    \item \textbf{音频数据读取}
    \begin{itemize}[leftmargin=3em]
        \item 从 I2S 接口读取 PDM 音频数据
        \item 将 PDM 数据转换为 PCM 数据（软件解码）
        \item 将音频数据缓存到内存中（使用环形缓冲区）
    \end{itemize}
    
    \item \textbf{录音控制逻辑}
    \begin{itemize}[leftmargin=3em]
        \item 实现按键触发录音（按下开始、松开停止）
        \item 或实现 VAD（语音活动检测），自动检测说话开始和结束
        \item 录音完成后生成 WAV 文件头（或直接裸音频数据），准备上传
    \end{itemize}
\end{enumerate}

\subsubsection*{达标要求}
\begin{itemize}
    \item[{\color{success}$\checkmark$}] 能通过硅麦采集到音频数据，串口打印出数值（有声音时数据明显变化）
    \item[{\color{success}$\checkmark$}] 能录制一段语音并保存为 WAV 格式（或裸 PCM 数据）
    \item[{\color{success}$\checkmark$}] 录制音频能通过 PC 播放验证（导出到电脑后可听）
\end{itemize}

\subsubsection*{参考代码结构}

\section{阶段五：云端 ASR + LLM + TTS 集成}
\textbf{预估：3 天 \quad 优先级：$\star\star\star\star\star$}

这是 AI 对话的核心逻辑链，需要把三个阶段二–四打通。

\subsubsection*{子任务清单}

\begin{enumerate}[leftmargin=2em, label=\textbf{5.\arabic*}]
    \item \textbf{云端服务选型}
    \begin{itemize}[leftmargin=3em]
        \item 确定语音识别（ASR）服务：如讯飞语音听写、百度语音识别、阿里语音识别
        \item 确定大语言模型（LLM）服务：如讯飞星火、百度文心一言、阿里通义千问、DeepSeek
        \item 确定语音合成（TTS）服务：如讯飞 TTS、百度 TTS、Edge TTS
        \item \textbf{推荐选型方案见下表}
    \end{itemize}
    
    \item \textbf{ASR 集成}
    \begin{itemize}[leftmargin=3em]
        \item 将阶段三录制的音频数据通过 HTTP POST 发送至 ASR 接口
        \item 解析返回的识别文本
        \item 处理好超时和错误情况
    \end{itemize}
    
    \item \textbf{LLM 集成}
    \begin{itemize}[leftmargin=3em]
        \item 将 ASR 识别文本作为 Prompt 发送至大模型 API
        \item 设置系统提示词（System Prompt），定义 AI 角色
        \item 解析返回的回复文本
    \end{itemize}
    
    \item \textbf{TTS 集成}
    \begin{itemize}[leftmargin=3em]
        \item 将 LLM 回复文本发送至 TTS 接口
        \item 接收返回的音频数据（MP3 或 PCM）
        \item 传递给阶段四的播放模块
    \end{itemize}
    
    \item \textbf{完整对话流程串联}
    \begin{itemize}[leftmargin=3em]
        \item 按下按键 $\rightarrow$ 录音 $\rightarrow$ ASR $\rightarrow$ LLM $\rightarrow$ TTS $\rightarrow$ 播放 $\rightarrow$ 等待下一次输入
        \item 在串口和屏幕上显示每个环节的状态
        \item 加入超时保护和错误重试机制
    \end{itemize}
\end{enumerate}

\subsubsection*{云端服务推荐方案}

\subsubsection*{达标要求}

\chapter{关键难点与风险提示}

\section{技术难点}

\begin{enumerate}[leftmargin=2em]
    \item \textbf{PDM 麦克风驱动}——ESP32-S3 的 I2S 在 PDM 模式下配置较为复杂，DMA 缓冲区大小和采样率需要反复调试。建议先用 Arduino 的 I2S 库跑通基础采集，再考虑性能优化。
    \item \textbf{音频数据上传}——WAV 文件较大（16kHz $\times$ 16bit $\approx$ 32KB/s），ESP32 内存有限，需合理分片上传或流式上传（Chunked Transfer）。
    \item \textbf{端到端延迟}——录音 $\rightarrow$ ASR $\rightarrow$ LLM $\rightarrow$ TTS $\rightarrow$ 播放，每个环节都会引入延迟。整体控制在 3--5 秒内为佳。
    \item \textbf{内存管理}——ESP32-S3 虽然有 PSRAM，但多个任务同时运行（Wi-Fi、I2S、显示、手柄）时要注意栈空间分配。
\end{enumerate}

\section{风险与应对}

\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabularx}{\textwidth}{|c|X|X|}
    \hline
    \heiti 风险 & \heiti 影响 & \heiti 应对措施\\
    \hline
    ASR 识别率低 & 对话体验差 & 先测试不同 ASR 服务，选用最合适的\\
    \hline
    API 请求超时 & 对话中断 & 加入超时重试机制，设置合理超时阈值\\
    \hline
    硅麦采集失败 & 无法输入语音 & 先用手头的 USB 麦克风在 PC 端做 ASR 测试，验证云端链路再调试硬件\\
    \hline
    Wi-Fi 不稳定 & 对话/手柄中断 & 加入自动重连，界面显示连接状态\\
    \hline
    扬声器音量小 & 听不清回复 & 检查功放模块供电和增益设置\\
    \hline
\end{tabularx}
\end{center}
